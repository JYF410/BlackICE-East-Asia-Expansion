# Communism on the Rise
country_event = {
	id = political.1
	title = political.1.t
	desc = political.1.d
	picture = GFX_report_event_british_communists
	
	is_triggered_only = yes

	trigger = {
		NOT = { has_government = communism }
		has_civil_war = no
	}
	
	# Seek government support
	option = {
		name = political.1.a
		trigger = {
			OR = {
				NOT = { tag = CHI }
				AND = {
					NOT = { country_exists = PRC }
					tag = CHI
				}			
			}
		}
		custom_effect_tooltip = political.1.a.tt
		clr_country_flag = communism_popular_movement
		set_country_flag = communism_political_movement
	}

	# Seek popular support
	option = {
		name = political.1.b
		custom_effect_tooltip = political.1.b.tt
		clr_country_flag = communism_political_movement
		clr_country_flag = communism_government_support
		set_country_flag = communism_popular_movement
	}
}

country_event = {
	id = political.300
	hidden = yes 
	timeout_days = 1
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_flag = political_3_run }
		has_country_flag = communism_government_support
		communism > 0.4
		NOT = { has_government = communism }
		has_civil_war = no
		is_major = no
		is_puppet = no
		has_idea_with_trait = communist_revolutionary
		OR = {
			NOT = { tag = CHI }
			AND = {
				NOT = { country_exists = PRC }
				tag = CHI
			}			
		}
	}
	
	immediate = {
		if = {
			limit = {
				communism > 0.5
			}
			country_event = { id = political.3 days = 50 }
			set_country_flag = { flag = political_3_run value = 1 days = 50 }
		}
		else = {
			country_event = { id = political.3 days = 100 }
			set_country_flag = { flag = political_3_run value = 1 days = 100 }
		}
	}
	
}
