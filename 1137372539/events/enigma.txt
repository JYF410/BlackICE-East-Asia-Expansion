		controls_state = 126
	}

		hidden_effect = {	
			country_event = { id = enigma.6 days = 270 }
		}
	}

		name = enigma.5.a
		# radar stations are state level building, maximum is 6, depending on tech level. State 129 is East Midlands
		129 = {
			add_building_construction = {
				type = radar_station
				level = 3
				instant_build = yes
			}
		}
		add_tech_bonus = {
			name = bletchey_park_focus
			uses = 1
			bonus = 0.5
			category = decryption_tech
		}
		if = {
			limit = {
				has_dlc = "La Resistance"
			}
			add_ideas = Bletchley_Park_lar
		}
		else = {
			add_ideas = Bletchley_Park
		}
	}
}
