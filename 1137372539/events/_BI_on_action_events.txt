country_event = {
	id = BI_on_action.4
	hidden = yes  #timeout
    timeout_days = 1

	is_triggered_only = yes
	
	immediate = {
		log = "[Root.GetTag] ROOT | [From.FROM.GetTag] FROM"
		if = {
			limit = {
				NOT = { is_in_array = { FROM.FROM.volunteers_from = ROOT } }
			}
			add_to_array = { FROM.FROM.volunteers_from = ROOT }
		}
		
		if = {
			limit = {
				has_volunteers_amount_from = {
					tag = ROOT
					count > 5
				}
			}
			add_to_temp_variable = { num_volunteers = 6 }
		}
		else_if = {
			limit = {
				has_volunteers_amount_from = {
					tag = ROOT
					count > 4
				}
			}
			add_to_temp_variable = { num_volunteers = 5 }
		}
		else_if = {
			limit = {
				has_volunteers_amount_from = {
					tag = ROOT
					count > 3
				}
			}
			add_to_temp_variable = { num_volunteers = 4 }
		}
		else_if = {
			limit = {
				has_volunteers_amount_from = {
					tag = ROOT
					count > 2
				}
			}
			add_to_temp_variable = { num_volunteers = 3 }
		}
		else_if = {
			limit = {
				has_volunteers_amount_from = {
					tag = ROOT
					count > 1
				}
			}
			add_to_temp_variable = { num_volunteers = 2 }
		}
		else_if = {
			limit = {
				has_volunteers_amount_from = {
					tag = ROOT
					count > 0
				}
			}
			add_to_temp_variable = { num_volunteers = 1 }
		}
		
		#log = "[?num_volunteers] num_volunteers"
		
		subtract_from_temp_variable = { num_volunteers = FROM.FROM.num_volunteers_from@ROOT }
		add_to_variable = { FROM.FROM.num_volunteers_from@ROOT = num_volunteers }
		#log = "[?FROM.FROM.num_volunteers_from@ROOT] FROM.FROM.num_volunteers_from@ROOT"
		
		FROM.FROM = { #SPR
			every_enemy_country = {
				for_loop_effect = {
					end = num_volunteers
					
					add_opinion_modifier = {
						target = ROOT
						modifier = volunteers_sent_to_enemy
					}
				}
				
				for_each_scope_loop = { #SPA
					array = volunteers_from
					
					#log = "[This.GetTag] inside VOLUNTEERS FROM.FROM of [Prev.GetTag] [?PREV.num_volunteers_from@THIS]"
					for_loop_effect = {
						end = num_volunteers
						#log = "apply towards ROOT [Root.GetTag]"
						
						add_opinion_modifier = {
							target = ROOT
							modifier = volunteers_sent_against_us
						}
					}
					
					set_temp_variable = { enemy_volunteers = PREV.num_volunteers_from@THIS }
					subtract_from_temp_variable = { enemy_volunteers = ROOT.enemy_volunteers@THIS }
					add_to_variable = { ROOT.enemy_volunteers@THIS = enemy_volunteers }
					
					for_loop_effect = {
						end = enemy_volunteers
						
						#log = "apply from ROOT towards THIS [This.GetTag]"
						reverse_add_opinion_modifier = {
							target = ROOT
							modifier = volunteers_sent_against_us
						}
					}
				}
			}
			
		}
		
	}
}

		}

			add_to_array = { ai_difficulty_setting_targets = ROOT.setting_add_country }
									
			clear_array = ROOT.ai_setting_target_countries

				array = ai_difficulty_setting_targets

			}
		}

	}
	
	option = {
		name = "Abort"

		clear_variable = setting_holder
	}
