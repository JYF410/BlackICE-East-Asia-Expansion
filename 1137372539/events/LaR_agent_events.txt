country_event = { #operation_sabotage_fuel
	id = lar_operative_event.100
	title = lar_operative_event.100.t
	desc = lar_operative_event.100.desc
	picture = GFX_report_event_ipc_oil_gfxaddon

	is_triggered_only = yes
	
	option = {
		name = Understood
		
		effect_tooltip = {
		
			var:operation_target_state = {
				meta_effect = {
					text = {
						damage_building = {
							type = fuel_silo
							damage = [D]
							repair_speed_modifier = [R]
						}
					}
					D = "[?FROM.operation_sabotage_fuel_damage@ROOT]"
					R = "[?FROM.operation_sabotage_fuel_repair@ROOT]"
				}
				if = {
					limit = {
						FROM = { has_country_flag = operation_sabotage_fuel_critical@ROOT }
					}
					damage_building = {
						type = infrastructure
						damage = 2
						repair_speed_modifier = -0.2
					}
				}
			}
			
		}
		
		clear_variable = operation_target_state
		
		FROM = {
			clear_variable = operation_sabotage_fuel_damage@ROOT
			clear_variable = operation_sabotage_fuel_repair@ROOT
			
			clr_country_flag = operation_sabotage_fuel_critical@ROOT
		}
	}
}

		name = Understood

				set_temp_variable = { effect = FROM.operation_sabotage_oil_damage@ROOT }
				set_temp_variable = { repair = FROM.operation_sabotage_oil_repair@ROOT }
				set_temp_variable = { critical = -0.25 }
				if = {
					limit = {
						FROM = { has_country_flag = operation_sabotage_oil_critical@ROOT }
					}
					set_temp_variable = { critical = -0.5 }
				}
			
				custom_effect_tooltip = SABOTAGE_OIL_TOOLTIP_EVENT
			}
			
			if = {
				limit = {
					NOT = { check_variable = { operation_spread_target_state = 0 } }
				}
				var:operation_spread_target_state = {
					set_temp_variable = { effect = FROM.operation_sabotage_oil_damage_spread@ROOT }
					set_temp_variable = { repair = FROM.operation_sabotage_oil_repair_spread@ROOT }
				
					custom_effect_tooltip = SABOTAGE_OIL_TOOLTIP_EVENT
				}
			}
		}
		
		clear_variable = operation_target_state
		clear_variable = operation_spread_target_state
		
		FROM = {
			clear_variable = operation_sabotage_oil_damage@ROOT
			clear_variable = operation_sabotage_oil_repair@ROOT
			clear_variable = operation_sabotage_oil_damage_spread@ROOT
