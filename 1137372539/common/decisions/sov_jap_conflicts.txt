							set_temp_variable = { attacker_ai = THIS }

						limit = {
							has_template = "Infantry Division"

							teleport_armies = { 
								limit = {
									tag = PREV
								}
							}
							add_dynamic_modifier = { modifier = ai_border_war days = 2 }
							create_unit = {
								division = "name = \"Infantry Division\" division_template = \"Infantry Division\" start_experience_factor = 0.6 start_equipment_factor = 1"
								owner = PREV
								count = 6
								id = 55
							}
						}
					}
				}
			}
			
			if = {
				limit = {
					var:defender_state_pup = {
						CONTROLLER = {
							is_ai = yes
							always = no
							set_temp_variable = { defender_ai = THIS }
						}
					}
				}
				var:defender_ai = {
					if = {
						limit = {
							has_template = "Infantry Division"
						}
						var:PREV.defender_state_pup = {
							teleport_armies = { 
								limit = {
									tag = PREV
								}
							}
							add_dynamic_modifier = { modifier = ai_border_war days = 2 }
							create_unit = {
								division = "name = \"Infantry Division\" division_template = \"Infantry Division\" start_experience_factor = 0.6 start_equipment_factor = 1"
								owner = PREV
								count = 6
								id = 55
							}
						}
					}
				}
			}
				
			start_border_war = {
				change_state_after_war = no
				attacker = {
					state = var:attacker_state_pup
					num_provinces = 2
					on_win = SOV_border_conflict_won_attacker.2
					on_lose = SOV_border_conflict_lost_attacker.2
					on_cancel = SOV_border_conflict_canceled_attacker.2
				}
				
				defender = {
					state = var:defender_state_pup
					num_provinces = 2
					on_win = SOV_border_conflict_won_defender.2
					on_lose = SOV_border_conflict_lost_defender.2
					on_cancel = SOV_border_conflict_canceled_defender.2
				}
			}
			set_border_war_data = {
				attacker = var:attacker_state_pup
				defender = var:defender_state_pup
				combat_width = 60 
			}
			FROM = {
				activate_targeted_decision = { target = ROOT decision = WTT_border_conflict_time_until_cancelled }
			}
			activate_targeted_decision = { target = FROM decision = WTT_border_conflict_time_until_cancelled }
			
			hidden_effect = {
			}
		}
