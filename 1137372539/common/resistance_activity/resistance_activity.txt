sabotage_arms_factory = {
	available = {
		non_damaged_building_level = {
			building = arms_factory
			level > 0
		}
	}

	weight = {
		base = 1
		modifier = {
			add = var:modifier@resistance_chance_buildings
		}
	}

	effect = {
		set_temp_variable = { d = 0.25 }
		set_temp_variable = { f = modifier@resistance_damage_to_buildings }
		add_to_temp_variable = { f = FROM.modifier@resistance_damage_to_buildings }
		add_to_temp_variable = { f = 1 }
		multiply_temp_variable = { d = f }
		meta_effect = {
			text = {
				damage_building = { type = arms_factory damage = [D] }
			}
			D = "[?d]"
		}
	}

	alert_text = arms_factory_is_sabotaged
}

	available = {
		has_resources_amount = {
			resource = oil
			amount > 0
			delivered = yes
		}
	}

	weight = {
		base = 1
		modifier = {
			add = var:modifier@resistance_chance_resource
		}
	}

	effect = {
		clear_sabotaged_resources_if_necesary = yes
		
		set_temp_variable = { d = 12 }
		set_temp_variable = { f = modifier@resistance_damage_to_resources }
		add_to_temp_variable = { f = FROM.modifier@resistance_damage_to_resources }
		add_to_temp_variable = { f = 1 }
		multiply_temp_variable = { d = f }
		subtract_from_variable = { sabotaged_oil = d }
		
		set_temp_variable = { duration = 180 }
		set_temp_variable = { f = modifier@resistance_damage_to_resources_duration }
		add_to_temp_variable = { f = FROM.modifier@resistance_damage_to_resources_duration }
		add_to_temp_variable = { f = 1 }
		multiply_temp_variable = { duration = f }
		
		add_dynamic_modifier = {
			modifier = sabotaged_resources
			days = duration
		}
	}

	alert_text = resource_is_sabotaged
}
