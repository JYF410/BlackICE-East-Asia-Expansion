
ai_buy_convoys_when_low_majors = {
	allowed = {
		has_dlc = "Arms Against Tyranny"
	}
	
	enable = {
		has_country_flag = coastal_state
		is_major = yes
		stockpile_ratio = {
			archetype = convoy
			ratio < 100
		}
	}
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = equipment_market_buy
		equipment_type = convoy
		value = 250
	}
}
