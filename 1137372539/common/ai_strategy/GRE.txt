GRE_guarantee_ROM = {
	allowed = {
		tag = GRE
	}
	enable = {
		country_exists = ROM
	}
	abort = {
		OR = {
			has_war_with = ROM
			OR = {
				AND = {
					is_in_faction_with = GER
					GER = {
						has_war_with = ROM
					}
				}
				AND = {
					is_in_faction_with = ENG
					ENG = {
						has_war_with = ROM
					}
				}
				AND = {
					is_in_faction_with = SOV
					SOV = {
						has_war_with = ROM
					}
				}
			}
			has_global_flag = end_of_the_balkan_pact
		}
	}
	ai_strategy = {
		type = protect
		id = "ROM"
		value = 200
	}
}
