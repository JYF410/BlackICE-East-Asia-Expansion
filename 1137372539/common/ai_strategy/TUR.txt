	allowed = { tag = TUR }
	enable = {
		has_war = no
	}
	abort = {
		OR = {
			has_war = yes
			ENG = { 
				has_war_with = GER 
				
				GER = { 
					OR = {
						surrender_progress > 0.25 
						has_capitulated = yes
					}
				}
			}
		}
	}
	
	ai_strategy = { type = alliance id = ENG value = -1000 }
	ai_strategy = { type = alliance id = USA value = -1000 }
}
