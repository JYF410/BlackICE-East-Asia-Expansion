difficulty_player = {
	if = { limit = { check_variable = { on_add = 1 } }
		
		if = {
			limit = {
				difficulty > 2
			}
			meta_effect = {
				text = {
					add_dynamic_modifier = {
						modifier = player_offset_[X]
					}
				}
				X = "[?global.difficulty]"
			}
			force_update_dynamic_modifier = yes
		}
		else = {
			custom_effect_tooltip = "Requires Veteran or BICE difficulty"
		}
		
	}
	else_if = { limit = { check_variable = { on_remove = 1 } }
		
		if = {
			limit = {
				difficulty > 2
			}
			meta_effect = {
				text = {
					remove_dynamic_modifier = {
						modifier = player_offset_[X]
					}
				}
				X = "[?global.difficulty]"
			}
			force_update_dynamic_modifier = yes
		}
		
	}
}
