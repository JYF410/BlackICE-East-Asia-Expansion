		}

		limit = { NOT = { has_idea = Arab_hyper_inflation_1 } }

PER_anger_ENG = {
	if = {
		limit = {
			check_variable = { ENG.PER_angry = 6 }
		}
		custom_effect_tooltip = PER_anger_ENG_tt_danger
	}
	else = {
		custom_effect_tooltip = PER_anger_ENG_tt
	}
	hidden_effect = {
		ENG = {
			add_to_variable = { PER_angry = 1 }
			if = {
				limit = {
					is_ai = yes
					check_variable = { PER_angry > 7 }
					not = { is_in_faction_with = PER }
					has_war_with = GER
					SOV = {
						has_war_with = GER
					}
					NOT = { has_war_with = PER }
					NOT = { SOV = { has_war_with = PER } }
				}
				if = {
					limit = {
						NOT = { has_war_with = PER }
					}
					PER = {
						add_timed_idea = {
							idea = last_hope
							days = 500
						}
						declare_war_on = {
							target = SOV
							type = annex_everything
						}
					}
				}
				SOV = {
					if = {
						limit = {
							is_ai = yes
							check_variable = { PER_angry > 4 }
							has_war_with = GER
							not = { has_non_aggression_pact_with = PER }
							not = { is_in_faction_with = PER }
						}
						if = {
							limit = {
								NOT = { has_war_with = PER }
							}
							PER = {
								declare_war_on = {
									target = ENG
									type = annex_everything
								}
							}
						}
					}
				}
			}
			else = {
				add_opinion_modifier = {
					target = PER
					modifier = kinda_angry
				}
			}		
		}
	}
}
