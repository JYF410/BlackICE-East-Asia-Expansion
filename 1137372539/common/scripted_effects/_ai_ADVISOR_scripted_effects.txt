ai_update_advisor_logic = {

	if = {
		limit = {
			NOT = { has_country_flag = ai_advisor_start_check }
		}
		set_country_flag = ai_advisor_start_check
		clear_array = ai_starting_advisor
		
		every_character = {
			limit = {
				is_advisor = yes
			}
			
			if = {
				limit = {
					is_character_slot = economic_minister
					is_hired = yes
				}
				add_to_array = { PREV.ai_starting_advisor@economic_minister = key }
				add_to_array = { PREV.ai_starting_advisor_scope@economic_minister = THIS }
			}
		}
		
	}
	
	if = {
		limit = {
			has_country_flag = ai_assign_advisor
		}
		clr_country_flag = ai_assign_advisor
		
		if = {
			limit = {
				NOT = { has_country_flag = ai_military_build_switch }
			}
			set_temp_variable = { ai_build_civilian = @AI_ECONOMIC_FOCUS_SCORE }
		}
		else = {
			set_temp_variable = { ai_build_military = @AI_ECONOMIC_FOCUS_SCORE }
		}
	
		if = {
			limit = {
				has_country_flag = ai_custom_advisor_ai
			}
			
			meta_effect = {
				text = {
					d_ai_advisor_template/[X] = yes
				}
				X = "[This.GetTag]"
			}
		}
		else = {
			d_ai_advisor_template/generic = yes
		}
			
		for_each_loop = {
			array = ai_starting_advisor@economic_minister
			
			if = {
				limit = {
					meta_trigger = {
						text = {
							has_idea = [X]_economic_minister
						}
						X = "[?v.GetTokenKey]"
					}
					NOT = { is_in_array = { ai_character@economic_minister = v } }
				}
				add_to_temp_array = { ai_character@economic_minister = v }
				meta_effect = {
					text = {			
						set_temp_variable = { [X]_economic_minister = -100 }
					}
					X = "[?v.GetTokenKey]"
				}
				#log = "set score to -100 for [?v.GetTokenKey]"
			}
			else = {
				add_to_temp_array = { remove_starting = v }
				#log = "[?v.GetTokenKey] REMOVE THIS GUY AS HE IS NO LONGER ASSIGNED!"
			}
		}
		for_each_loop = { array = remove_starting remove_from_array = { ai_starting_advisor@economic_minister = v } }
		
		ai_select_next_advisor_slot_target = yes
		
		remove_ideas = {
			ai_advisor_1
			ai_advisor_2
			ai_advisor_3
		}
		
		if = {
			limit = {
				check_variable = { advisor_picked = 0 }
			}
			add_political_power = 150
			country_event = { id = ai_MISC.3 days = 30 } #delayed removal
		}
		else = {
			remove_ideas = ai_advisor
		}
	}
}

ai_select_next_advisor_slot_target = {
	
	for_each_loop = {
		array = global.idea_slots
		value = slot
		break = none
		index = entry
		
		if = { #WORKAROUND FOR TOKEN: ISSUE
			limit = {
				OR = {
					check_variable = { slot = token:tank_manufacturer }
					check_variable = { slot = token:naval_manufacturer }
					check_variable = { slot = token:aircraft_manufacturer }
					check_variable = { slot = token:materiel_manufacturer }
					check_variable = { slot = token:industrial_concern }
					check_variable = { slot = token:electrical_concern }
				}
			}
		}
		else = {
			for_each_loop = {
				array = ai_character@var:slot
				
				add_to_temp_array = { ai_character_temp@var:slot = v }
			}
			clear_temp_array = ai_character@var:slot
			for_each_loop = {
				array = ai_character_temp@var:slot
				
				meta_effect = {
					text = {
						set_temp_variable = { t = [X] }
					}
					X = "[?v.GetTokenKey]"
				}
				add_to_temp_array = { ai_character@var:slot = t }
			}
		}
		
		if = {
			limit = {
				OR = {
					check_variable = { ai_character@var:slot^num = 0 }
					check_variable = { ai_character@var:slot^num > 1 } #if theres more than the max then evaluate them, no matter if slot is filled or not
					AND = {
						check_variable = { ai_character@var:slot^num = 1 } #if theres only one available guy, no need to run any code if hes already assigned
						if = {
							limit = {
								OR = {
									check_variable = { slot = token:tank_manufacturer }
									check_variable = { slot = token:materiel_manufacturer }
									check_variable = { slot = token:naval_manufacturer }
									check_variable = { slot = token:aircraft_manufacturer }
									check_variable = { slot = token:industrial_concern }
									check_variable = { slot = token:electrical_concern }
								}
							}
							NOT = {
								has_idea = var:ai_character@var:slot^0
							}
						}
						else_if = {
							limit = {
								set_temp_variable = { t = ai_character@var:slot^0 }
								
								set_temp_variable = { no_character_available = 1 }
								has_character = var:t
								set_temp_variable = { no_character_available = 0 }
								
								var:t = {
									set_temp_variable = { token_key = key }
								}
							}
							NOT = {
								meta_trigger = {
									text = { has_idea = [X]_[Y] }
									X = "[?token_key.GetTokenKey]"
									Y = "[?slot.GetTokenKey]"
								}
							}
						}
					}
				}
			}
			if = {
				limit = {
					check_variable = { no_character_available = 1 }
				}
				clear_temp_array = ai_character@var:slot
			}
			
			set_temp_variable = { ai_score = ai_score_base@var:slot }
			
			#-------------------------------------------------------------------------
			multiply_temp_variable = { ai_score = ai_score_mult@var:slot?1 }
			add_to_temp_variable = { ai_score = ai_score_add@var:slot }
			
			randomize_temp_variable = { var = random_score distribution = uniform min = 0 max = ai_advisor_slot_max_random_score }
			add_to_temp_variable = { ai_score = random_score }

			add_to_temp_array = { ai_advisor_slot_token = slot }
			add_to_temp_array = { ai_advisor_slot_max = global.idea_slots_num^entry }
			add_to_temp_array = { ai_advisor_slot_score = ai_score }
		}
	}
	
	ai_sort_advisor_slot_scores = yes
}

ai_sort_advisor_slot_scores = { 

	#ADVISOR_log = "=========================================="
	#ADVISOR_log = "=================[This.GetTag]==================="
	### FIND HIGHEST SCORED SLOT 
	
	for_each_loop = { 
		array = ai_advisor_slot_score
		break = large_break
		
		set_temp_variable = { link = ai_advisor_slot_token^i }
		set_temp_variable = { max = ai_advisor_slot_max^i }
		
		for_each_loop = { array = ai_advisor_slot_score_sorted
			value = not_v

			if = { limit = { check_variable = { v > not_v } }

				add_to_temp_array = { array = ai_advisor_slot_token_sorted index = i value = link }
				add_to_temp_array = { array = ai_advisor_slot_max_sorted index = i value = max }
				add_to_temp_array = { array = ai_advisor_slot_score_sorted index = i value = v }
				set_temp_variable = { break = 1 }
			}
		}

		if = { limit = { check_variable = { break = 0 } }
			
			add_to_temp_array = { array = ai_advisor_slot_token_sorted value = link }
			add_to_temp_array = { array = ai_advisor_slot_max_sorted value = max }
			add_to_temp_array = { array = ai_advisor_slot_score_sorted value = v }
		}
		else = { set_temp_variable = { break = 0 } }
	}
	
	
	### FIND HIGHEST SCORED ADVISOR OF HIGHEST SCORED SLOT, AND THEN CHECK IF ITS AVAILABLE TRIGGER IS TRUE
	
	#log = "---> ALL SLOT SCORES <---"

	if = {
		limit = {
			has_global_flag = ai_track_advisors
		}
		clear_array = ai_advisor_slot_debug
		clear_array = ai_advisor_slot_score_debug
		for_each_loop = {
			array = ai_advisor_slot_token_sorted
			
			#log = "SLOT: [?v.GetTokenKey] | SCORE: [?ai_advisor_slot_score_sorted^i]"
			add_to_array = { ai_advisor_slot_debug = v }
			add_to_array = { ai_advisor_slot_score_debug = ai_advisor_slot_score_sorted^i }
		}
	}
	#log = " "
	#log = " "
	
	for_each_loop = {
		array = ai_advisor_slot_token_sorted
		value = slot
		
		if = {
			limit = {
				check_variable = { ai_advisor_slot_score_sorted^i > @AI_MIN_ADVISOR_SLOT_SCORE }
			}
			
			set_temp_variable = { max_slot = ai_advisor_slot_max_sorted^i }
			set_temp_variable = { slot_num_filled = 0 }
			
			for_each_loop = { #ADD NEWLY PICKED ADVISORS TO THE TEMP ARRAY
				array = ai_character_new@var:slot
				
				add_to_temp_array = { ai_character@var:slot = v }
			}
			
			#ADVISOR_log = " "
			#ADVISOR_log = "SLOT: [?slot.GetTokenKey] MAX: [?max_slot] | SCORE: [?ai_advisor_slot_score_sorted^i] | EVAL: [?ai_character@var:slot^num]"
			
			if = {
				limit = {
					NOT = {
						check_variable = { slot = token:industrial_concern }
						check_variable = { slot = token:electrical_concern }
						check_variable = { slot = token:materiel_manufacturer }
						check_variable = { slot = token:aircraft_manufacturer }
						check_variable = { slot = token:naval_manufacturer }
						check_variable = { slot = token:tank_manufacturer }
					}
				}
				set_temp_variable = { is_character = 1 }
			}
			else = { set_temp_variable = { is_character = 0 } }
			
			clear_temp_array = ai_advisor_token
			clear_temp_array = ai_advisor_score
			clear_temp_array = ai_advisor_assigned
			
			clear_temp_array = ai_advisor_token_sorted
			clear_temp_array = ai_advisor_score_sorted
			clear_temp_array = ai_advisor_assigned_sorted
			
			clear_temp_array = remove_invalid_advisor
			
			### GET ADVISOR SCORES
			
			if = { #CHARACTER
				limit = {
					check_variable = { is_character = 1 }
				}
				for_each_loop = {
					array = ai_character@var:slot 
					if = {
						limit = { #check if the character exists and has the correct character slot
							has_character = var:v
							var:v = {
								meta_trigger = {
									text = {
										is_character_slot = [Y]
									}
									Y = "[?slot.GetTokenKey]"
								}
							}
						}
						var:v = { set_temp_variable = { token_key = key } }
						
						meta_effect = { #add the manual score 
							text = {
								set_temp_variable = { score = [Y]_[X] }
							}
							X = "[?slot.GetTokenKey]"
							Y = "[?token_key.GetTokenKey]"
						}
						
						
						if = {
							limit = { #if advisor is assigned, score is increased
								meta_trigger = {
									text = {
										has_idea = [X]_[Y]
									}
									X = "[?token_key.GetTokenKey]"
									Y = "[?slot.GetTokenKey]"
								}
							}
							add_to_temp_variable = { score = @AI_ASSIGNED_ADVISOR_SCORE }
							set_temp_variable = { assigned_status = 1 }
							
							add_to_temp_array = { assigned_advisor_token@var:slot = token_key }
							add_to_temp_array = { assigned_advisor_value@var:slot = v }
						}
						else = { set_temp_variable = { assigned_status = 0 } }
						
						add_to_temp_array = { ai_advisor_token = v }
						add_to_temp_array = { ai_advisor_score = score }
						add_to_temp_array = { ai_advisor_assigned = assigned_status }
					}
					else = { #if character is not valid anymore add it to temp array for removal
					
						add_to_temp_array = { remove_invalid_advisor = v }
					}
				}
						
			### IF NOT ENOUGH ADVISOR IS PRESENT -> FIND NEW ONE
			
				ai_get_new_advisor_for_slot = yes
				
			}
			
			else = { #NON CHARACTER
			
				for_each_loop = {
					array = ai_character@var:slot
					
					meta_effect = { #add the manual score 
						text = {
							set_temp_variable = { score = [Y]_[X] }
						}
						X = "[?slot.GetTokenKey]"
						Y = "[?v.GetTokenKey]"
					}
					
					if = {
						limit = { #if designer is assigned, score is increased
							has_idea = var:v
						}
						add_to_temp_variable = { score = @AI_ASSIGNED_ADVISOR_SCORE }
						set_temp_variable = { assigned_status = 1 }
					}
					else = { set_temp_variable = { assigned_status = 0 } }
						
					add_to_temp_array = { ai_advisor_token = v }
					add_to_temp_array = { ai_advisor_score = score }
					add_to_temp_array = { ai_advisor_assigned = assigned_status }
				}
			}
			
			if = { #remove invalid characters from the list
				limit = {
					check_variable = { remove_invalid_advisor^num > 0 }
				}
				
				for_each_loop = {
					array = remove_invalid_advisor
					value = remove_value
					
					remove_from_temp_array = { ai_character@var:slot = remove_value }
					if = {
						limit = {
							is_in_array = { ROOT.ai_character_new@var:slot = remove_value }
						}
						remove_from_array = { ROOT.ai_character_new@var:slot = remove_value }
					}
				}
			}
				
		### FIND AND SORT HIGHEST SCORED ADVISOR 
			
			if = {
				limit = {
					check_variable = { ai_advisor_token^num > 0 }
				}
					
				for_each_loop = { 
					array = ai_advisor_score
					break = large_break
					
					set_temp_variable = { link = ai_advisor_token^i }
					set_temp_variable = { link2 = ai_advisor_assigned^i }
					
					for_each_loop = { array = ai_advisor_score_sorted
						value = not_v

						if = { limit = { check_variable = { v > not_v } }

							add_to_temp_array = { array = ai_advisor_token_sorted index = i value = link }
							add_to_temp_array = { array = ai_advisor_score_sorted index = i value = v }
							add_to_temp_array = { array = ai_advisor_assigned_sorted index = i value = link2 }
							set_temp_variable = { break = 1 }
						}
					}

					if = { limit = { check_variable = { break = 0 } }
						
						add_to_temp_array = { array = ai_advisor_token_sorted value = link }
						add_to_temp_array = { array = ai_advisor_score_sorted value = v }
						add_to_temp_array = { array = ai_advisor_assigned_sorted value = link2 }
					}
					else = { set_temp_variable = { break = 0 } }
				}
			}
			
			
		### CHECK HIGHEST SCORED ADVISOR IF THEY ARE AVAILABLE
			
			if = {
				limit = {
					OR = {
						check_variable = { slot = token:economic_minister }
						check_variable = { slot = token:domestic_minister }
					}
				}
				for_each_loop = {
					array = ai_advisor_token_sorted
					
					#log = "[?v.GetName] [?ai_advisor_score_sorted^i] "
					if = {
						limit = {
							is_in_array = { assigned_advisor_value@var:slot = v }
						}
						var:v = { set_temp_variable = { lowest_advisor = key } }
					}
					
				}
			}
			else = { set_temp_variable = { lowest_advisor = 0 } }
			
			for_each_loop = {
				array = ai_advisor_token_sorted
				
				ai_select_next_advisor = yes
				
			}
		}
	}

ai_get_new_advisor_for_slot = {

	set_temp_variable = { best_weight = 0 }
	set_temp_variable = { best_weight_token = 0 }

	if = {
		limit = {
			OR = {
				check_variable = { ai_character@var:slot^num = 0 }
				check_variable = { ai_character@var:slot^num < max_slot }
			}
		}
		
		if = {
			limit = {
				check_variable = { slot = token:high_command }
			}
			
			for_each_loop = {
				array = ai_character@high_command
				value = exists
				
				var:exists = {
					if = {
						limit = {
							has_army_ledger = yes
						}
						add_to_temp_variable = { potential_high_command_army = 1 }
					}
					else_if = {
						limit = {
							has_air_ledger = yes
						}
						add_to_temp_variable = { potential_high_command_air = 1 }
					}
					else_if = {
						limit = {
							has_navy_ledger = yes
						}
						add_to_temp_variable = { potential_high_command_navy = 1 }
					}
				}
			}
			
			every_character = {
				limit = {
					#is_advisor = yes
					is_character_slot = high_command
					has_variable = key
					
					if = {
						limit = {
							has_army_ledger = yes
						}
						check_variable = { potential_high_command_army < 2 }
					}
					else_if = {
						limit = {
							has_air_ledger = yes
						}
						check_variable = { potential_high_command_air < 2 }
					}
					else_if = {
						limit = {
							has_navy_ledger = yes
						}
						check_variable = { potential_high_command_navy < 2 }
					}
					NOT = { 
						is_in_array = { ai_character@var:slot = THIS } 
					}
				}
				
				set_temp_variable = { weight = mtth:new_advisor_ai_will_do }
				
				if = { limit = { check_variable = { weight > best_weight } }
					set_temp_variable = { best_weight = weight }
					set_temp_variable = { best_weight_token = THIS }
				}
				
			}
		}
		else_if = {
			limit = {
				check_variable = { slot = token:theorist }
			}
			
			for_each_loop = {
				array = ai_character@theorist
				value = exists
				
				var:exists = {
					if = {
						limit = {
							has_army_ledger = yes
						}
						add_to_temp_variable = { potential_theorist_army = 1 }
					}
					else_if = {
						limit = {
							has_air_ledger = yes
						}
						add_to_temp_variable = { potential_theorist_air = 1 }
					}
					else_if = {
						limit = {
							has_navy_ledger = yes
						}
						add_to_temp_variable = { potential_theorist_navy = 1 }
					}
				}
			}
			
			every_character = {
				limit = {
					#is_advisor = yes
					is_character_slot = theorist
					has_variable = key
				
					if = {
						limit = {
							has_army_ledger = yes
						}
						check_variable = { potential_theorist_army < 2 }
					}
					else_if = {
						limit = {
							has_air_ledger = yes
						}
						check_variable = { potential_theorist_air < 2 }
					}
					else_if = {
						limit = {
							has_navy_ledger = yes
						}
						check_variable = { potential_theorist_navy < 2 }
					}
					NOT = { 
						is_in_array = { ai_character@var:slot = THIS } 
					}
				}
				
				set_temp_variable = { weight = mtth:new_advisor_ai_will_do }
				
				if = { limit = { check_variable = { weight > best_weight } }
					set_temp_variable = { best_weight = weight }
					set_temp_variable = { best_weight_token = THIS }
				}
				
			}
		}
		else = {
			every_character = {
				limit = {
					is_advisor = yes 
					
					meta_trigger = { 
						text = {
							is_character_slot = [X]
						}
						X = "[?slot.GetTokenKey]"
					}
					
					has_variable = key
					NOT = { is_in_array = { ai_character@var:slot = THIS } }
				}
				
				set_temp_variable = { weight = mtth:new_advisor_ai_will_do }
					
				if = { limit = { check_variable = { weight > best_weight } }
					set_temp_variable = { best_weight = weight }
					set_temp_variable = { best_weight_token = THIS }
				}
			}
		}
		
		### ADD NEW BEST SCORED ADVISOR TO ARRAY
		if = { 
			limit = { 
				check_variable = { best_weight > 0 } 
				#check_variable = { best_weight_token > 0 } 
			}
			#ADVISOR_log = "NEW CHARACTER [?best_weight_token.GetName] [?best_weight_token] FOR [?slot.GetTokenKey] FOR [ROOT.GetTag]"
				
			add_to_array = { ROOT.ai_character_new@var:slot = best_weight_token }
			
			add_to_temp_array = { ai_advisor_token = best_weight_token }
			add_to_temp_array = { ai_advisor_score = best_weight }
			add_to_temp_array = { ai_advisor_assigned = 0 }
		}
			
	}

ai_select_next_advisor = {
	
	if = {
		limit = {
			check_variable = { is_character = 1 }
		}
		#ADVISOR_log = "    ADVISOR: [?v.GetName] | SCORE: [?ai_advisor_score_sorted^i]"
	}
	else = {
		#ADVISOR_log = "    IDEA: [?v.GetTokenKey] | SCORE: [?ai_advisor_score_sorted^i]"
	}
	if = {
		limit = {
			check_variable = { slot_num_filled < max_slot }
				
			NOT = { check_variable = { ai_advisor_score_sorted^i < @AI_MIN_ADVISOR_SCORE } }
		}
		
		## CHARACTER
		if = {
			limit = {
				check_variable = { is_character = 1 }
				
				if = {
					limit = {
						check_variable = { ai_advisor_assigned_sorted^i = 1 }
					}
					add_to_temp_variable = { slot_num_filled = 1 }
					#ADVISOR_log = "        -already assigned"
					always = no
					
				}
				
				if = {
					limit = {
						check_variable = { slot = token:high_command }
					}
					if = {
						limit = {
							var:v = { has_army_ledger = yes }
							ai_can_pick_high_command_army = no
						}
						#ADVISOR_log = "        -ai_can_pick_high_command_ARMY false"
						always = no
					}
					if = {
						limit = {
							var:v = { has_air_ledger = yes }
							ai_can_pick_high_command_air = no
						}
						#ADVISOR_log = "        -ai_can_pick_high_command_AIR false"
						always = no
					}
					if = {
						limit = {
							var:v = { has_navy_ledger = yes }
							ai_can_pick_high_command_navy = no
						}
						#ADVISOR_log = "        -ai_can_pick_high_command_NAVY false"
						always = no
					}
				}
				if = {
					limit = {
						check_variable = { slot = token:theorist }
					}
					if = {
						limit = {
							var:v = { has_army_ledger = yes }
							ai_can_pick_theorist_army = no
						}
						#ADVISOR_log = "        -ai_can_pick_theorist_ARMY false"
						always = no
					}
					if = {
						limit = {
							var:v = { has_air_ledger = yes }
							ai_can_pick_theorist_air = no
						}
						#ADVISOR_log = "        -ai_can_pick_theorist_AIR false"
						always = no
					}
					if = {
						limit = {
							var:v = { has_navy_ledger = yes }
							ai_can_pick_theorist_navy = no
						}
						#ADVISOR_log = "        -ai_can_pick_theorist_NAVY false"
						always = no
					}
				}
				
			}
			
			var:v = {
				meta_effect = {
					text = {
						add_trait = {
							trait = ai_trait
							slot = [X]
						}
					}
					X = "[?slot.GetTokenKey]"
				}
			}
			
			if = { ### CHECK IF ADVISOR IS AVAILABLE
				limit = {
					has_available_idea_with_traits = { idea = ai_trait limit = 1 characters = yes }
				}
				#ADVISOR_log = " "
				#ADVISOR_log = "----> FINAL AI PICK: [?v.GetName] | [?slot.GetTokenKey] | IS CHARACTER"
					
				if = {
					limit = {
						has_global_flag = ai_track_advisors
					}
					
					if = {
						limit = {
							is_in_array = { ROOT.ai_advisor_history = v }
						}
						log = "[Root.GetTag] ADDED AGAIN [?v.GetTokenKey] [?v.GetName]"
					}
					add_to_array = { ROOT.ai_advisor_history = v }
					add_to_array = { ROOT.ai_advisor_history_slot = slot }
					add_to_array = { ROOT.ai_advisor_history_date = global.date }
				}
				
				if = { 
					limit = {
						check_variable = { lowest_advisor > 0 }
						OR = {
							AND = {
								check_variable = { slot = token:economic_minister }
								has_country_flag = { flag = has_economic_minister value = 2 }
							}
							AND = {
								check_variable = { slot = token:domestic_minister }
								has_country_flag = { flag = has_domestic_minister value = 2 }
							}
						}
					}
					
					log = "[Root.GetTag] | Remove to make room [?slot.GetTokenKey] [?lowest_advisor.GetTokenKey] [?slot.GetTokenKey]"
					
					meta_effect = {
						text = {
							remove_ideas = [Y]_[X]
						}
						X = "[?slot.GetTokenKey]"
						Y = "[?lowest_advisor.GetTokenKey]"
					}
				}
					
				var:v = {
					meta_effect = {
						text = {
							remove_trait = {
								trait = ai_trait
								slot = [X]
							}
							ROOT = { add_ideas = [Y]_[X] }
						}
						X = "[?slot.GetTokenKey]"
						Y = "[?key.GetTokenKey]"
					}
				}
				
				set_temp_variable = { advisor_picked = 1 }
				
				set_temp_variable = { break = 1 }
			}
			else = {
				var:v = {
					meta_effect = {
						text = {
							remove_trait = {
								trait = ai_trait
								slot = [X]
							}
						}
						X = "[?slot.GetTokenKey]"
					}
					#ADVISOR_log = "        -unavailable TRIGGER"
				}
			}
				
		}
		
		## NON CHARACTER
		else_if = { ### CHECK IF NON CHARACTER IDEA IS AVAILABLE
			limit = {
				NOT = { check_variable = { is_character = 1 } }
				
				if = {
					limit = {
						check_variable = { ai_advisor_assigned_sorted^i = 1 }
					}
					add_to_temp_variable = { slot_num_filled = 1 }
					#ADVISOR_log = "        -already assigned"
					always = no
				}
				meta_trigger = {
					text = {
						has_available_idea_with_traits = { idea = [X]_trait limit = 1 characters = no }
					}
					X = "[?slot.GetTokenKey]"
				}
			}
		
			#ADVISOR_log = "----> FINAL AI PICK: [?v.GetTokenKey] | [?slot.GetTokenKey] | IS IDEA"
			
			if = {
				limit = {
					has_global_flag = ai_track_advisors
				}
				if = {
					limit = {
						is_in_array = { ROOT.ai_advisor_history = v }
					}
					log = "[Root.GetTag] ADDED AGAIN [?v.GetTokenKey] [?slot.GetTokenKey]"
				}
				add_to_array = { ROOT.ai_advisor_history = v }
				add_to_array = { ROOT.ai_advisor_history_slot = slot }
				add_to_array = { ROOT.ai_advisor_history_date = global.date }
			}
			
			add_ideas = var:v #this doesnt work for advisors
			
			set_temp_variable = { advisor_picked = 1 }
			
			set_temp_variable = { break = 1 }
		}
	}
	
}
