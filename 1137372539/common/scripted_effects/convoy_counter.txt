get_highest_port_provinces_coastal = {
	get_coastal_provinces = yes
	
	set_temp_variable = { building = token:naval_base }
	
	#log = "[?coastal_provinces^num] coastal_provinces"
	for_each_loop = {
		array = coastal_provinces
		
		count_province_building = yes
		if = {
			limit = {
				check_variable = { province_level > highest_province_level }
			}
			set_temp_variable = { highest_province_level = province_level }
			set_temp_variable = { province = v }
		}
	}
	
	#log = "[?province] province | [?highest_province_level] highest_province_level"
	
	if = {
		limit = {
			check_variable = { province > 0 }
		}
		
		remove_from_temp_array = { coastal_provinces = province }
		
		set_temp_variable = { province_level = 0 }
		set_temp_variable = { highest_province_level = 0 }
		for_each_loop = {
			array = coastal_provinces
		
			count_province_building = yes
			if = {
				limit = {
					check_variable = { province_level > second_highest_province_level }
				}
				set_temp_variable = { second_highest_province_level = province_level }
				set_temp_variable = { second_province = v }
			}
		}
		#log = "[?second_province] second_province | [?second_highest_province_level] second_highest_province_level"
	}
}

generate_get_coastal_provinces = {
	every_state = {
		limit = { is_coastal = yes }
		clear_array = coastal_provinces
		set_building_level = {
			type = naval_base #naval_base_test
			level = 3
			province = {
				all_provinces = yes
				limit_to_coastal = yes
			}
		}
		log = "else_if = { limit = { state = [This.GetID] }"
		
		for_each_loop = {
			array = global.province_controllers
			
			if = {
				limit = {
					meta_trigger = {
						text = {
							any_province_building_level = {
								province = {
									id = [PROVINCE]
								}
								building = naval_base #naval_base_test
								level > 2
							}
						}
						PROVINCE = "[?i]"
					}
				}
				log = "    add_to_temp_array = { coastal_provinces = [?i] }"
				#add_to_array = { coastal_provinces = i }
			}
		}
		log = "}"
		
		set_building_level = {
			type = naval_base ##naval_base_test
			level = 0
			province = {
				all_provinces = yes
				limit_to_coastal = yes
			}
		}
	}
	
	
			
}
