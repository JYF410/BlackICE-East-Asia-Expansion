update_lack_of_fuel = {

	set_temp_variable = { days_storage_target = civilian_fuel_use }
	multiply_temp_variable = { days_storage_target = 24 }
	multiply_temp_variable = { days_storage_target = 90 }
	divide_temp_variable = { days_storage_target = 1000 }
	
	set_temp_variable = { max_value = max_fuel_k }
	clamp_temp_variable = { var = days_storage_target max = max_value }
	
	set_variable = { fuel_min = days_storage_target }
		
	if = {
		limit = {
			OR = {
				check_variable = { days_storage_target > fuel_k }
				check_variable = { lack_of_fuel > 0 }
			}
		}
	
		if = {
			limit = {
				NOT = { is_in_array = { alerts = token:alert_fuel_deficit } }
			}
			add_to_array = { alerts = token:alert_fuel_deficit }
			add_to_array = { alerts_sender = THIS }
		}

		set_temp_variable = { t = fuel_k }
		divide_temp_variable = { t = days_storage_target }
		
		set_variable = { lack_of_fuel_target = 1 }
		subtract_from_variable = { lack_of_fuel_target = t }
		clamp_variable = { var = lack_of_fuel_target min = 0 }
		
		if = { #growth
			limit = {
				check_variable = { lack_of_fuel_target > lack_of_fuel }
			}
			
			set_temp_variable = { growth = 0.01 }
			
			add_to_variable = { lack_of_fuel = growth }
			clamp_variable = { var = lack_of_fuel max = lack_of_fuel_target }
		}
		else = { #decay
			limit = {
				check_variable = { lack_of_fuel_target < lack_of_fuel }
			}
			set_temp_variable = { decay = 0.02 }
			
			subtract_from_variable = { lack_of_fuel = decay }
			clamp_variable = { var = lack_of_fuel min = lack_of_fuel_target }
		}
		
		clamp_variable = { var = lack_of_fuel min = 0 max = 1 }
		
		#################################
		# update modifiers 
	
		if = { limit = { NOT = { has_dynamic_modifier = { modifier = lack_of_fuel } } }
			add_dynamic_modifier = { modifier = lack_of_fuel }
		}
		
		if = {
			limit = {
				check_variable = { lack_of_fuel < 0.333 }
			}
			
			set_temp_variable = { scale = lack_of_fuel }
			divide_temp_variable = { scale = 0.333 }
			
			lack_of_fuel_1 = yes
		}
		else_if = {
			limit = {
				check_variable = { lack_of_fuel > 0.666 }
			}
			
			set_temp_variable = { scale = lack_of_fuel }
			subtract_from_temp_variable = { scale = 0.666 }
			divide_temp_variable = { scale = 0.333 }
			
			lack_of_fuel_3 = yes
		}
		else = {
			set_temp_variable = { scale = lack_of_fuel }
			subtract_from_temp_variable = { scale = 0.333 }
			divide_temp_variable = { scale = 0.333 }
			
			lack_of_fuel_2 = yes
		}
	}
	else_if = {
		limit = {
			has_dynamic_modifier = { modifier = lack_of_fuel }
		}
		remove_dynamic_modifier = { modifier = lack_of_fuel }
		clear_variable = lofu_stab
		clear_variable = lofu_drift
		clear_variable = lofu_prod
		clear_variable = lofu_prodn
		for_each_loop = {
			array = alerts
			value = alert_v
			index = alert_i
			
			if = {
				limit = {
					check_variable = { alert_v = token:alert_fuel_deficit }
				}
				remove_from_array = { array = alerts_sender index = alert_i }
			}
		}
		remove_from_array = { alerts = token:alert_fuel_deficit }
	
	}
}

calculate_equipment_supplied_ratio = {
	
	for_each_loop = {
		array = global.archetypes_infantry
		
		if = {
			limit = {
				check_variable = { num_target_equipment_in_armies_k@var:v > 0 }
			}
				
			set_temp_variable = { t = num_equipment_in_armies_k@var:v }
			clamp_temp_variable = { var = t max = num_target_equipment_in_armies_k@var:v }
			
			add_to_temp_variable = { equipment_fielded = t }
			add_to_temp_variable = { equipment_stock = num_equipment@var:v }
			add_to_temp_variable = { equipment_fielded_target = num_target_equipment_in_armies_k@var:v }
			
		}
	}
	if = {
		limit = {
			check_variable = { num_armor > 0 }
		}
		for_each_loop = {
			array = global.archetypes_armor
			
			if = {
				limit = {
					check_variable = { num_target_equipment_in_armies_k@var:v > 0 }
				}
					
				set_temp_variable = { t = num_equipment_in_armies_k@var:v }
				clamp_temp_variable = { var = t max = num_target_equipment_in_armies_k@var:v }
				
				add_to_temp_variable = { equipment_fielded = t }
				add_to_temp_variable = { equipment_stock = num_equipment@var:v }
				add_to_temp_variable = { equipment_fielded_target = num_target_equipment_in_armies_k@var:v }
				
			}
		}
	}

	divide_temp_variable = { equipment_stock = 1000 }
	divide_temp_variable = { equipment_stock = equipment_fielded }
	
	divide_temp_variable = { equipment_fielded = equipment_fielded_target }
	
	if = {
		limit = {
			check_variable = { equipment_stock = 0 }
		}
		clear_variable = stock_ratio
	}
	else = {
		set_variable = { stock_ratio = equipment_stock }
	}
	set_variable = { field_strength = equipment_fielded }
	
}

	if = {
		limit = {

			}
		}
		add_to_temp_variable = { base_storage = 250 }
		set_variable = { storage_base = 250 }
	}
	else = {
		clear_variable = storage_base
	}
	
	for_each_loop = {
		array = global.storage
			
		set_temp_variable = { storage_modifier = ROOT.modifier@max_resource_storage_factor }
		add_to_temp_variable = { storage_modifier = 1 }
		multiply_temp_variable = { storage_modifier = global.resource_cost@var:v }
		multiply_temp_variable = { storage_modifier = 50 }
		
		set_variable = { max_storage@var:v = num_storage@var:v }
		multiply_variable = { max_storage@var:v = storage_modifier }
		add_to_variable = { max_storage@var:v = base_storage }
		
		set_temp_variable = { max_storage_resource@var:v = resource_produced@var:v }
		add_to_temp_variable = { max_storage_resource@var:v = export_offset@var:v }
		multiply_temp_variable = { max_storage_resource@var:v = 3 }
		add_to_variable = { max_storage@var:v = max_storage_resource@var:v }
		
		if = {
			limit = {
				check_variable = { storage_output@var:v > 0 }
			}
			set_temp_variable = { factor = modifier@storage_output_factor }
			add_to_temp_variable = { factor = 1 }
			
			set_variable = { storage_output@var:v = global.resource_cost@var:v }
			multiply_variable = { storage_output@var:v = factor }
			set_temp_variable = { t = storage_output@var:v }
			multiply_variable = { storage_output@var:v = num_storage@var:v }
			add_to_variable = { storage_output@var:v = t }
		}
	}

}
state_count_storage_buildings = {
	if = {
		limit = {
			check_variable = { building_level@storage > 0 }
		}
		
		set_temp_variable = { check = 0 }
		for_each_loop = {
			array = global.storage
			 
			add_to_variable = { ROOT.num_storage@var:v = storage@var:v }
			add_to_temp_variable = { check = storage@var:v }
		}
		
		if = {
			limit = {
				check_variable = { building_level@storage > check }
			}
			set_variable = { assign_storage = building_level@storage }
			subtract_from_variable = { assign_storage = check }
			
			add_to_array = { ROOT.assign_storage_state = THIS }
			
			add_to_variable = { ROOT.assign_storage = assign_storage }
		}
		else = {
			log = "Warning! There is more assigned storage than actual storage in [This.GetName]"
		}
	}
}

				remove_opinion_modifier = {
					target = PREV
					modifier = overlord_opinion_trade
				}

		}

		}
		
		remove_ideas = civil_war

		limit = {
			is_subject = yes 
			NOT = { 
				OVERLORD = {
					PREV = {
						has_country_flag = applied_overlord_opinion@PREV
					}
				}
			}
		}
        OVERLORD = {
			PREV = {
				set_country_flag = applied_overlord_opinion@PREV
			}
            add_opinion_modifier = {
                target = PREV
                modifier = overlord_opinion
            }
			PREV = {
				add_opinion_modifier = {
					target = PREV
					modifier = overlord_opinion_trade
				}
			}
        }
	}
	
	if = {
		limit = {
			NOT = { check_variable = { modifier@import_cost_return = 0 } }
		}
		set_temp_variable = { t = resource_imported@tungsten }
		add_to_temp_variable = { t = resource_imported@chromium }
		add_to_temp_variable = { t = resource_imported@rubber }

		add_to_temp_variable = { ic_for_imports = t }
		
		set_temp_variable = { t = resource_imported@aluminium }
		add_to_temp_variable = { t = resource_imported@oil }
		multiply_temp_variable = { t = 0.1 }
		add_to_temp_variable = { ic_for_imports = t }
		
		set_temp_variable = { t = resource_imported@steel }
		multiply_temp_variable = { t = 0.083 }
		add_to_temp_variable = { ic_for_imports = t }
		
		set_temp_variable = { t = resource_imported@bauxite }
		add_to_temp_variable = { t = resource_imported@iron }
		multiply_temp_variable = { t = 0.04 }
		add_to_temp_variable = { ic_for_imports = t }
		
		set_temp_variable = { t = resource_imported@coal }
		multiply_temp_variable = { t = 0.031 }
		add_to_temp_variable = { ic_for_imports = t }
		
		set_temp_variable = { t = resource_imported@food }
		multiply_temp_variable = { t = 0.05 }
		add_to_temp_variable = { ic_for_imports = t }
		
		multiply_temp_variable = { ic_for_imports = modifier@import_cost_return }
		add_to_temp_variable = { ic_for_imports = 0.5 }
		clamp_temp_variable = { var = ic_for_imports max = modifier@max_import_cost_return }
		round_temp_variable = ic_for_imports
		
		subtract_from_temp_variable = { ic_for_imports = import_return }
		add_to_variable = { import_return = ic_for_imports }
		
		add_offsite_building = { 
			type = industrial_complex 
			level = var:ic_for_imports 
		}
		
	}
	else_if = {
		limit = {
			has_variable = import_return
		}
		multiply_variable = { import_return = -1 }
		add_offsite_building = { 
			type = industrial_complex 
			level = var:import_return
		}
		clear_variable = import_return
	}
}

get_offensive_or_defensive_war_status = {
	for_each_scope_loop = {
		array = enemies 
		
		if = { limit = { PREV = { has_offensive_war_with = PREV } }
			add_to_temp_variable = { num_offensive_war = 1 }
		}
		else_if = { limit = { PREV = { has_defensive_war_with = PREV } }
			add_to_temp_variable = { num_defensive_war = 1 }
		}
	}
	if = { 
		limit = {
			has_war = yes 
		}
		if = {
			limit = { 
				check_variable = { num_offensive_war > num_defensive_war } 
				NOT = { has_idea = offensive_war }
			}
			
			if = { limit = { has_idea = defensive_war }
				remove_ideas = defensive_war
			}
			
			add_ideas = offensive_war
		}
		else_if = {
			limit = { 
				#check_variable = { num_defensive_war > num_offensive_war } 
				NOT = { has_idea = defensive_war }
			}
			
			if = { limit = { has_idea = offensive_war }
				remove_ideas = offensive_war
			}
			
			add_ideas = defensive_war
		}
	}
}
