## Order is important in case of a tie
SPR_seize_power = {
	if = {
		limit = { 
			NOT = {
				OR = {
					has_country_flag = commie_take_power
					has_country_flag = CNTFAI_take_power
					has_country_flag = libs_take_power
				}
			}
		}
		if = {
			limit = { 
				check_variable = { com_inf > 99 } 
				}
			hidden_effect = {
				set_politics = {
					ruling_party = communism
				}
				set_country_flag = commie_take_power
			}
		}
		else_if = {
			limit = { 
				check_variable = { ana_inf > 99 } 
				}
			hidden_effect = {
				set_politics = {
					ruling_party = socialism
				}
				set_country_flag = CNTFAI_take_power 
			}
		}
		else_if = {
			limit = { 
				check_variable = { lib_inf > 99 } 
				}
			hidden_effect = {
				set_politics = {
					ruling_party = liberalism
				}
				set_country_flag = libs_take_power
			}
		}
	}
}

improve_postwar = {
	if = {
		limit = {
			has_idea = SPR_postwar1
		}
		swap_ideas = {
			add_idea = SPR_postwar2
			remove_idea = SPR_postwar1
		}
	}
	else_if = {
		limit = {
			has_idea = SPR_postwar2
		}
		swap_ideas = {
			add_idea = SPR_postwar3
			remove_idea = SPR_postwar2
		}
	}
	else_if = {
		limit = {
			has_idea = SPR_postwar3
		}
		swap_ideas = {
			add_idea = SPR_postwar4
			remove_idea = SPR_postwar3
		}
	}
	else_if = {
		limit = {
			has_idea = SPR_postwar4
		}
		swap_ideas = {
			add_idea = SPR_postwar5
			remove_idea = SPR_postwar4
		}
	}	
	else_if = {
		limit = {
			has_idea = SPR_postwar5
		}
		remove_ideas = SPR_postwar5
		add_timed_idea = { idea = SPR_postwar6 days = 180 }
	}
}
