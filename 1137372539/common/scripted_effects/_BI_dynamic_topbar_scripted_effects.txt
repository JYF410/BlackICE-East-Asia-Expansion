update_topbar_menu_category = {
	
	if = {
		limit = {
			check_variable = { is_sidebar = 1 }
		}
		set_variable = { side_subcategory_selected = topbar_menu_v }
		
		#log = "update category"
		set_temp_variable = { next_index = topbar_menu_i }
		for_each_loop = {
			array = side_subcategory
			
			add_to_temp_variable = { next_index = 3 }
			
			#log = "run subcategory [?v.GetTokenKey]"
			if = {
				limit = {
					check_variable = { sidebar_menu@var:sidebar_i^num < next_index }
				}
				add_to_array = { sidebar_menu@var:sidebar_i = 0 }
				add_to_array = { sidebar_menu@var:sidebar_i = 0 }
				add_to_array = { sidebar_menu@var:sidebar_i = 0 }
			}
			
			add_to_array = { array = sidebar_menu@var:sidebar_i value = v index = next_index }
			
			set_temp_variable = { remove_shift = next_index }
			add_to_temp_variable = { remove_shift = 1 }
			remove_from_array = { array = sidebar_menu@var:sidebar_i index = remove_shift }
			
		}
	}
	else = {
		set_variable = { subcategory_selected = topbar_menu_v }
		
		#log = "update category"
		set_temp_variable = { next_index = topbar_menu_i }
		for_each_loop = {
			array = subcategory
			
			add_to_temp_variable = { next_index = 3 }
			
			#log = "run subcategory [?v.GetTokenKey]"
			if = {
				limit = {
					check_variable = { topbar_menu@var:topbar_i^num < next_index }
				}
				add_to_array = { topbar_menu@var:topbar_i = 0 }
				add_to_array = { topbar_menu@var:topbar_i = 0 }
				add_to_array = { topbar_menu@var:topbar_i = 0 }
			}
			
			add_to_array = { array = topbar_menu@var:topbar_i value = v index = next_index }
			
			set_temp_variable = { remove_shift = next_index }
			add_to_temp_variable = { remove_shift = 1 }
			remove_from_array = { array = topbar_menu@var:topbar_i index = remove_shift }
			
		}
	}
}

update_topbar_menu = {
	create_customizeable_topbar_sidebar_content = yes
	
	if = {
		limit = {
			check_variable = { is_sidebar = 1 }
		}
		clear_array = sidebar_menu@var:sidebar_i
		
		for_each_loop = {
			array = global.topbar_content
			
			add_to_array = { sidebar_menu@var:sidebar_i = v }
		}
		
		if = {
			limit = {
				has_tech = guards_infantry_tech
			}
			add_to_array = { sidebar_menu@var:sidebar_i = token:topbar_guards }
		}
	}
	else = {
		clear_array = topbar_menu@var:topbar_i
		
		for_each_loop = {
			array = global.topbar_content
			
			add_to_array = { topbar_menu@var:topbar_i = v }
		}
		
		if = {
			limit = {
				has_tech = guards_infantry_tech
			}
			add_to_array = { topbar_menu@var:topbar_i = token:topbar_guards }
		}
	}
}
